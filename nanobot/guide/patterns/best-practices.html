异步编程 ​

使用 asyncio ​

python
# ✅ 推荐：使用 async/await
async def process_message(msg: InboundMessage):
    response = await self.provider.chat(messages)
    await self.bus.publish_outbound(response)

# ❌ 避免：阻塞操作
async def process_message(msg: InboundMessage):
    # 这会阻塞事件循环！
    result = blocking_function()

并行执行 ​

python
# ✅ 并行执行多个独立任务
results = await asyncio.gather(
    task1(),
    task2(),
    task3()
)

# ❌ 顺序执行（不必要的等待）
result1 = await task1()
result2 = await task2()
result3 = await task3()

超时控制 ​

python
# ✅ 设置超时
try:
    result = await asyncio.wait_for(
        long_running_task(),
        timeout=5.0
    )
except asyncio.TimeoutError:
    logger.error("任务超时")

错误处理 ​

捕获具体异常 ​

python
# ✅ 捕获具体异常
try:
    with open(path) as f:
        content = f.read()
except FileNotFoundError:
    return f"Error: 文件不存在: {path}"
except PermissionError:
    return f"Error: 没有权限: {path}"

# ❌ 捕获所有异常
try:
    with open(path) as f:
        content = f.read()
except Exception:  # 太宽泛
    return "Error"

错误消息返回 ​

python
# ✅ 返回有用的错误消息
async def execute(self, path: str) -> str:
    try:
        return await do_operation(path)
    except Exception as e:
        return f"Error: {self.__class__.__name__}: {str(e)}"

# LLM 可以根据错误信息尝试其他方法

资源清理 ​

python
# ✅ 使用 try/finally 或 async with
async def process(self):
    resource = await acquire_resource()
    try:
        result = await do_work(resource)
        return result
    finally:
        await release_resource(resource)

配置优先级 ​

python
# 优先级：环境变量 > 配置文件 > 默认值
def get_config():
    # 1. 尝试环境变量
    if "NANOBOT__model" in os.environ:
        return os.environ["NANOBOT__model"]

    # 2. 尝试配置文件
    config = load_config()
    if config.model:
        return config.model

    # 3. 使用默认值
    return "claude-sonnet-4-20250514"

代码风格 ​

使用类型注解 ​

python
# ✅ 推荐：类型注解
async def process_message(
    msg: InboundMessage,
    session: Session
) -> OutboundMessage | None:
    pass

# ❌ 不推荐：缺少类型
async def process_message(msg, session):
    pass

文档字符串 ​

python
# ✅ 推荐：详细的文档字符串
async def execute(self, path: str, encoding: str = "utf-8") -> str:
    """
    读取文件内容

    Args:
        path: 文件路径（相对于 workspace）
        encoding: 文件编码，默认 utf-8

    Returns:
        文件内容

    Raises:
        FileNotFoundError: 文件不存在
        PermissionError: 没有读取权限

    Example:
        >>> tool = ReadFileTool(workspace=Path("/tmp"))
        >>> await tool.execute("test.txt")
        "Hello, World!"
    """

性能优化 ​

缓存 ​

python
# ✅ 缓存频繁访问的数据
class SessionManager:
    def __init__(self):
        self._cache: dict[str, Session] = {}

    def get_or_create(self, key: str) -> Session:
        if key in self._cache:
            return self._cache[key]  # 缓存命中
        # 加载并缓存
        session = self._load(key) or Session(key)
        self._cache[key] = session
        return session

流式处理 ​

python
# ✅ 处理大文件时使用流式
async def read_large_file(path: str) -> AsyncIterator[str]:
    async with aiofiles.open(path) as f:
        async for line in f:
            yield line

# ❌ 一次性读取大文件
async def read_large_file(path: str) -> str:
    async with aiofiles.open(path) as f:
        return await f.read()  # 可能占用大量内存

常见陷阱 ​

阻塞事件循环 ​

python
# ❌ 阻塞调用
time.sleep(1)  # 阻塞整个事件循环

# ✅ 异步等待
await asyncio.sleep(1)

遗忘 await ​

python
# ❌ 遗忘 await
result = async_function()  # 返回 coroutine 对象，不是结果

# ✅ 正确使用
result = await async_function()

资源泄露 ​

python
# ✅ 确保资源释放
async with aiofiles.open(path) as f:
    content = await f.read()
# 文件自动关闭

# ❌ 可能泄露
f = await aiofiles.open(path)
content = await f.read()
# 如果这里出错，文件不会关闭
